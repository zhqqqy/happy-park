func getLength(url string) (int64, error) {
	resp, err := http.Get(url)
	if err != nil {
		return 0, err
	}
	length, err := strconv.ParseInt(resp.Header.Get("Content-Length"), 10, 64)

	if err != nil {
		return 0, err
	}
	return length, err
}
